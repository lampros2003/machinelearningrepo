def f0(x):
    """Probability density function for H0"""
    return 1 / np.sqrt(2 * np.pi) * np.exp(-(x**2) / 2)


def f1(x):
    """Probability density function for H1"""
    return 0.5 * (norm.pdf(x, -1, 1) + norm.pdf(x, 1, 1))


def bayes_test(x1, x2, simplified="simple"):
    """Optimal Bayes test"""
    if simplified == "simple":
        log_likelihood_ratio = np.log(np.cosh(x1) * np.cosh(x2))
        if log_likelihood_ratio > np.log(np.e):
            return 1  # Decide H1
        else:
            return 0  # Decide H0

    log_likelihood_ratio = np.log(f1(x1) / f0(x1)) + np.log(f1(x2) / f0(x2))
    if log_likelihood_ratio > 0:
        return 1  # Decide H1
    else:
        return 0  # Decide H0

def generate_data(n_samples, hypothesis):
    """Generate  for H0 or H1"""
    if hypothesis == 0:
        # H0: N(0,1)
        return np.random.normal(0, 1, (n_samples, 2))
    else:
        # H1: 0.5{N(-1,1) + N(1,1)}
        means = np.random.choice([-1, 1], size=(n_samples, 2))
        return np.random.normal(means, 1)


def initialize_parameters(input_size, hidden_size, output_size):
    W1 = np.random.normal(
        0, np.sqrt(1 / (input_size + hidden_size)), (input_size, hidden_size)
    )
    b1 = np.zeros((1, hidden_size))
    W2 = np.random.normal(
        0, np.sqrt(1 / (hidden_size + output_size)), (hidden_size, output_size)
    )
    b2 = np.zeros((1, output_size))
    return {"W1": W1, "b1": b1, "W2": W2, "b2": b2}


def forward_pass(X, parameters):
    W1, b1 = parameters["W1"], parameters["b1"]
    W2, b2 = parameters["W2"], parameters["b2"]
    # First layer
    Z1 = np.dot(X, W1) + b1
    A1 = np.tanh(Z1)  # Using tanh as activation for hidden layer
    # Output layer
    Z2 = np.dot(A1, W2) + b2
    cache = {"Z1": Z1, "A1": A1, "Z2": Z2}
    return cache

class ADAMer:
    def __init__(self, learning_rate=0.001, beta1=0.9, beta2=0.999, epsilon=1e-8):
        self.learning_rate = learning_rate
        self.beta1 = beta1
        self.beta2 = beta2
        self.epsilon = epsilon
        self.m = {}
        self.v = {}
        self.t = 0

    def initialize(self, parameters):
        for key in parameters.keys():
            self.m[key] = np.zeros_like(parameters[key])
            self.v[key] = np.zeros_like(parameters[key])

    def update(self, parameters, gradients):
        self.t += 1

        for key in parameters.keys():
            # Update biased first moment estimate
            self.m[key] = self.beta1 * self.m[key] + (1 - self.beta1) * gradients[key]

            # Update biased second raw moment estimate
            self.v[key] = self.beta2 * self.v[key] + (1 - self.beta2) * np.square(
                gradients[key]
            )

            # Compute bias-corrected first moment estimate
            m_hat = self.m[key] / (1 - self.beta1**self.t)

            # Compute bias-corrected second raw moment estimate
            v_hat = self.v[key] / (1 - self.beta2**self.t)
            # Update parameters
            parameters[key] -= (
                self.learning_rate * m_hat / (np.sqrt(v_hat) + self.epsilon)
            )

def train_network(
    X_train_h0,
    X_train_h1,
    method="cross_entropy",
    epochs=100,
    convergence_threshold=1e-4,
):

    input_size = 2
    hidden_size = 20
    output_size = 1

    # Initialize parameters
    parameters = initialize_parameters(input_size, hidden_size, output_size)
    optimizer = ADAMer(learning_rate=0.001)
    optimizer.initialize(parameters)

    n_samples = len(X_train_h0)

    # Track both costs and errors
    cost_history = []
    error_history = []
    smoothed_costs = []
    smoothed_errors = []

    def compute_cost_and_error(x0_sample, x1_sample, params):
        cache_h0 = forward_pass(x0_sample, params)
        cache_h1 = forward_pass(x1_sample, params)
        ##### Evaluation of errors, costs
        if method == "cross_entropy":
            output_h0 = sigmoid(cache_h0["Z2"])
            output_h1 = sigmoid(cache_h1["Z2"])

            cost_h0 = -np.log(1 - output_h0 + 1e-10)
            cost_h1 = -np.log(output_h1 + 1e-10)
            # Error for cross-entropy
            error_h0 = (output_h0 >= 0.5).mean()
            error_h1 = (output_h1 < 0.5).mean()
        else:  # exponential method
            output_h0 = cache_h0["Z2"]
            output_h1 = cache_h1["Z2"]
            cost_h0 = np.exp(0.5 * output_h0)
            cost_h1 = np.exp(-0.5 * output_h1)

            # Error  for exponential
            error_h0 = (output_h0 >= 0).mean()
            error_h1 = (output_h1 < 0).mean()

        total_cost = np.mean(cost_h0) + np.mean(
            cost_h1
        )  # Average cost across both hypotheses
        total_error = 0.5 * (
            error_h0 + error_h1
        )  # Average error rate across both hypotheses

        return total_cost, total_error
        ##############

    #####
    def get_smoothed_metric(values):
        window_size = min(20, len(values))
        return np.mean(values[-window_size:])

    for epoch in range(epochs):
        # Shuffle training data
        indices = np.random.permutation(n_samples)
        X_h0_shuffled = X_train_h0[indices]
        X_h1_shuffled = X_train_h1[indices]

        epoch_costs = []
        epoch_errors = []

        for i in range(n_samples):
            X_h0_sample = X_h0_shuffled[i : i + 1]
            X_h1_sample = X_h1_shuffled[i : i + 1]

            # Forward pass
            cache_h0 = forward_pass(X_h0_sample, parameters)
            cache_h1 = forward_pass(X_h1_sample, parameters)

            # Compute gradients based on method
            if method == "cross_entropy":
                output_h0 = sigmoid(cache_h0["Z2"])
                output_h1 = sigmoid(cache_h1["Z2"])

                dZ2_h0 = output_h0 / (1 - output_h0 + 1e-10)
                dZ2_h1 = -1 / (output_h1 + 1e-10)
            else:
                output_h0 = cache_h0["Z2"]
                output_h1 = cache_h1["Z2"]
                dZ2_h0 = 0.5 * np.exp(0.5 * output_h0)
                dZ2_h1 = -0.5 * np.exp(-0.5 * output_h1)

            # Initialize gradients
            gradients = {
                "W1": np.zeros_like(parameters["W1"]),
                "b1": np.zeros_like(parameters["b1"]),
                "W2": np.zeros_like(parameters["W2"]),
                "b2": np.zeros_like(parameters["b2"]),
            }

            # Compute gradient hypotheses
            for dZ2, cache, X in [
                (dZ2_h0, cache_h0, X_h0_sample),
                (dZ2_h1, cache_h1, X_h1_sample),
            ]:
                # 
                gradients["W2"] += np.dot(cache["A1"].T, dZ2)
                gradients["b2"] += np.sum(dZ2, axis=0, keepdims=True)

                # Hidden 
                dA1 = np.dot(dZ2, parameters["W2"].T)
                dZ1 = dA1 * (1 - np.square(cache["A1"]))
                gradients["W1"] += np.dot(X.T, dZ1)
                gradients["b1"] += np.sum(dZ1, axis=0, keepdims=True)

            optimizer.update(parameters, gradients)

            # Compute cost and error for this sample pair
            sample_cost, sample_error = compute_cost_and_error(
                X_h0_sample, X_h1_sample, parameters
            )
            epoch_costs.append(sample_cost)
            epoch_errors.append(sample_error)

        # Calculate average metrics
        avg_epoch_cost = np.mean(epoch_costs)
        avg_epoch_error = np.mean(epoch_errors)

        cost_history.append(avg_epoch_cost)
        error_history.append(avg_epoch_error)

        # Calculate smoothed

        current_smoothed_cost = get_smoothed_metric(cost_history)
        current_smoothed_error = get_smoothed_metric(error_history)

        smoothed_costs.append(current_smoothed_cost)
        smoothed_errors.append(current_smoothed_error)

        # Check for convergence using smoothed costs
        if len(smoothed_costs) > 1:

            cost_change = abs(smoothed_costs[-1] - smoothed_costs[-2])

            if cost_change < convergence_threshold:
                print(f"Converged at epoch {epoch}")
                print(f"Final smoothed cost: {current_smoothed_cost:.4f}")
                print(f"Final smoothed error: {current_smoothed_error:.4f}")

                return parameters, {
                    "raw_costs": cost_history,
                    "smoothed_costs": smoothed_costs,
                    "errors": error_history,
                    "smoothed_errors": smoothed_errors,
                }

        # Progress logging
        if epoch % 10 == 0:
            print(f"Epoch {epoch}:")
            print(
                f"  Raw Cost: {avg_epoch_cost:.4f}, Smoothed Cost: {current_smoothed_cost:.4f}"
            )
            print(
                f"  Raw Error: {avg_epoch_error:.4f}, Smoothed Error: {current_smoothed_error:.4f}"
            )

    print("Maximum epochs reached without convergence")
    return parameters, {
        "raw_costs": cost_history,
        "smoothed_costs": smoothed_costs,
        "errors": error_history,
        "smoothed_errors": smoothed_errors,
    }

# Calculate final test error
def evaluate_test_error(X_test_h0, X_test_h1, parameters, method="cross_entropy"):
    # Combine X_test_h0 and X_test_h1 into a single array
    X_test = np.concatenate((X_test_h0, X_test_h1))

    # Create labels for the combined array
    y_test = np.concatenate((np.zeros(len(X_test_h0)), np.ones(len(X_test_h1))))

    # Shuffle the combined array and labels
    indices = np.arange(len(X_test))
    np.random.shuffle(indices)
    X_test = X_test[indices]
    y_test = y_test[indices]

    # Rest of the code remains the same
    n_samples = len(X_test)
    test_errors = []

    for i in range(n_samples):
        X_sample = X_test[i : i + 1]
        y_sample = y_test[i : i + 1]

        # Forward pass
        cache = forward_pass(X_sample, parameters)

        # Calculate errors (pred should match the labels)
        if method == "cross_entropy":
            pred = sigmoid(cache["Z2"]) > 0.5
        else:  # exponential method
            pred = cache["Z2"] > 0

        error = int(pred)  # false positive rate

        # Calculate accuracy
        accuracy = int(pred == y_sample)

        test_errors.append(accuracy)

    # Calculate average test accuracy
    avg_test_accuracy = np.mean(test_errors)

    return 1 - avg_test_accuracy

def exponential_loss(y_true, y_pred):
    return tf.reduce_mean(tf.exp(tf.abs(y_true - y_pred)))
